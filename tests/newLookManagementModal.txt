/* ---------- animowany licznik ---------- */

const CountUp: React.FC<{ value: number; duration?: number }> = ({ value, duration = 900 }) => {
  const [n, setN] = useState(0);
  useEffect(() => {
    let raf = 0;
    const start = performance.now();
    const tick = (t: number) => {
      const p = Math.min((t - start) / duration, 1);
      setN(value * (1 - Math.pow(1 - p, 3)));
      if (p < 1) raf = requestAnimationFrame(tick);
    };
    raf = requestAnimationFrame(tick);
    return () => cancelAnimationFrame(raf);
  }, [value, duration]);
  return <>{Math.round(n)}</>;
};

/* ---------- SVG: radar atrybutów (skala 1–20, N osi) ---------- */

const AttrRadar: React.FC<{
  attrs: { key: string; value: number }[];
  hovered: string | null;
}> = ({ attrs, hovered }) => {
  const [mounted, setMounted] = useState(false);
  useEffect(() => {
    const t = setTimeout(() => setMounted(true), 200);
    return () => clearTimeout(t);
  }, []);

  const n = attrs.length;
  if (n < 3) return null; // radar ma sens od 3 osi

  const C = 80, R = 62;
  const angle = (i: number) => -Math.PI / 2 + (i * 2 * Math.PI) / n;
  const px = (i: number, r: number) => C + Math.cos(angle(i)) * r;
  const py = (i: number, r: number) => C + Math.sin(angle(i)) * r;
  const poly = attrs.map((a, i) => `${px(i, (a.value / 20) * R)},${py(i, (a.value / 20) * R)}`).join(' ');

  return (
    <svg viewBox="0 0 160 160" className="w-full h-full" aria-hidden>
      <defs>
        <linearGradient id="mm-radar" x1="0" y1="0" x2="1" y2="1">
          <stop offset="0%" stopColor="#facc15" />
          <stop offset="100%" stopColor="#34d399" />
        </linearGradient>
      </defs>
      {[0.33, 0.66, 1].map((g) => (
        <polygon
          key={g}
          points={attrs.map((_, i) => `${px(i, R * g)},${py(i, R * g)}`).join(' ')}
          fill="none" stroke="#e0f2fe" strokeOpacity="0.1" strokeWidth="1"
        />
      ))}
      {attrs.map((_, i) => (
        <line key={i} x1={C} y1={C} x2={px(i, R)} y2={py(i, R)} stroke="#e0f2fe" strokeOpacity="0.07" />
      ))}
      <g style={{ transformOrigin: '80px 80px', transform: mounted ? 'scale(1)' : 'scale(0)', transition: 'transform 800ms cubic-bezier(.2,1.4,.4,1)' }}>
        <polygon points={poly} fill="url(#mm-radar)" fillOpacity="0.22" stroke="url(#mm-radar)" strokeWidth="2" strokeLinejoin="round" />
        {attrs.map((a, i) => (
          <circle
            key={a.key}
            cx={px(i, (a.value / 20) * R)}
            cy={py(i, (a.value / 20) * R)}
            r={hovered === a.key ? 6 : 3.5}
            fill={attrColor(a.value)}
            stroke="#050b14"
            strokeWidth="1.5"
            style={{ transition: 'r 180ms ease' }}
          />
        ))}
      </g>
    </svg>
  );
};

              </div>

      </div>
    </div>
  );
};

/** Pasek atrybutu z animowanym wypełnieniem */
const AttrBarFill: React.FC<{ value: number; delay: number }> = ({ value, delay }) => {
  const [mounted, setMounted] = useState(false);
  useEffect(() => {
    const t = setTimeout(() => setMounted(true), delay);
    return () => clearTimeout(t);
  }, [delay]);
  const color = attrColor(value);
  return (
    <div
      className="h-full rounded-full"
      style={{
        width: mounted ? `${(value / 20) * 100}%` : '0%',
        background: `linear-gradient(90deg, ${color}99, ${color})`,
        boxShadow: `0 0 8px ${color}66`,
        transition: 'width 900ms cubic-bezier(.25,1,.3,1)',
      }}
    />
  );
};
